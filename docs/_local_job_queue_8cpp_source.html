40bool LocalJobQueue::stealJobs(std::vector<std::unique_ptr<LocalJobQueue>>& threadQueues, std::function<void()>& job)
41{
42 // Need atleast 2 worker threads present if we wanna steal jobs, since 1 worker thread obviously cant steal from itself. The main thread is not a worker thread so isn't taken into account here.
43 if (threadQueues.size() < 2)
44 {
45 return false; // No other threads to steal from
46 }
47

51

54 continue; // Skip the current thread's queue
55 }
56
57 // Try to pop a job from the other thread's queue) and steal it for this worker thread. We use a reference to the job so we can execute it in the current thread after stealing it
